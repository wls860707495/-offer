# 包含min函数的栈
## 题目描述
定义栈的数据结构，请在该类型中实现一个能够得到栈中所含最小元素的min函数（时间复杂度应为O(1))。
## 解题思路
```
# -*- coding:utf-8 -*-
class Solution:
    def __init__(self):
        self.stack = []
        self.minvalue = []
    def push(self, node):
        # write code here
        min = self.min()
        if not min or node < min:
            self.minvalue.append(node)
        else:
            self.minvalue.append(min)
        self.stack.append(node)
    def pop(self):
        # write code here
        if self.stack:
            self.minvalue.pop()
            return self.stack.pop()
    def top(self):
        # write code here
        if self.stack:
            return self.stack[-1]
    def min(self):
        # write code here
        if self.minvalue:
            return self.minvalue[-1]

```
