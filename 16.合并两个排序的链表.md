# 合并两个排序的链表
## 题目描述
输入两个单调递增的链表，输出两个链表合成后的链表，当然我们需要合成后的链表满足单调不减规则。
## 解题思路
### 新建链表
```
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    # 返回合并后列表
    def Merge(self, pHead1, pHead2):
        # write code here
        pHead = ListNode(0)
        tmp = pHead
        while pHead1 != None and pHead2 != None:
            if pHead1.val<=pHead2.val:
                tmp.next = pHead1
                pHead1 = pHead1.next
            else:
                tmp.next = pHead2
                pHead2 = pHead2.next
            tmp = tmp.next
        if pHead1 == None:
            tmp.next = pHead2
        elif pHead2 == None:
            tmp.next = pHead1
        return pHead.next
```
### 在原有链表基础上进行改动
```
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    # 返回合并后列表
    def Merge(self, pHead1, pHead2):         
        # write code here 
        if pHead1 == None: 
            return pHead2
        if pHead2 == None:
            return pHead1 
        if pHead1.val <= pHead2.val:              //node1为头节点小的那一个链表，node2为头节点大的那个
            node1 = pHead1
            node2 = pHead2
        else:
            node1 = pHead2
            node2 = pHead1
        node = node1
        while node2 != None:                      //node1不可能为none，node1为none时会将next指向node2，故以node2为循环条件
            if node.val <= node2.val:
                if node1.next == None:
                    node1.next = node2
                    return node
                elif node1.next.val >= node2.val:
                    node2next = node2.next
                    node2.next = node1.next
                    node1.next = node2
                    node2 = node2next
                    node1 = node1.next
                else:
                    node1 = node1.next
        return node      
```
